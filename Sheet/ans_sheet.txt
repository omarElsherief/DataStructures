int Stack::Bottom()
{
    if (isEmpty())
        return -1;
    Node *temp = top;
    while (temp->next != NULL)
    {
        temp = temp->next;
    }
    return temp->data;
}

int Stack::Size()
{
    int size = 0;
    if (isEmpty())
        return 0;
    size++;
    Node *temp = top;
    while (temp->next != NULL)
    {
        temp = temp->next;
        size++;
    }
    return size;
}

int Stack::Top()
{
    return top->data;
}
